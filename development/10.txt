// 更新页面标题
function updatePageTitle() {
  const baseTitle = "鲁虺文化新闻资讯资源网 http://news.luhui.net 海量新闻资讯尽在掌握全球最好网真互动式种源文化传播交流索引平台 lǔ 鲁 huǐ 虺 wén 文 huà 化 wǎng 网- 敏锐洞察行业动态，深度挖掘新思维";
  const titleElement = document.getElementById('page-title');
  
  if (!titleElement) return;
  
  // 如果正在查看特定文章
  if (state.currentArticle) {
    titleElement.textContent = `${state.currentArticle.title} - ${baseTitle}`;
    // 更新搜索链接
    const searchLink = document.getElementById('article-search-link');
    if (searchLink) {
      searchLink.href = `http://search.luhui.net/?q=${encodeURIComponent(state.currentArticle.title)}`;
      searchLink.textContent = state.currentArticle.title;
    }
  } 
  // 如果在特定分类下
  else if (state.currentCategory && state.currentCategory !== 'all') {
    const categoryName = config.categories[state.currentCategory] || state.currentCategory;
    titleElement.textContent = `${categoryName} - ${baseTitle}`;
  } 
  // 首页或默认状态
  else {
    titleElement.textContent = baseTitle;
  }
}

以上修改实现了两个主要功能：

1. 动态标题更新：
   - 当浏览特定分类时，标题会显示"分类名称 - 基础标题"
   - 当查看特定文章时，标题会显示"文章标题 - 基础标题"
   - 其他情况下显示原始基础标题

2. 搜索链接添加：
   - 在文章详情页面中添加了"搜索看看"链接
   - 链接指向搜索页面，自动带上当前文章标题作为搜索关键词
