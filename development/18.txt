        // 检测分类中的文章数量
        async function detectArticleCount(ai) {
            let count = 0;
            let consecutiveFailures = 0;
            
            // 从ID=1开始尝试检测
            for (let id = 1; id <= MAX_DETECTION_ATTEMPTS; id++) {
                try {
                    const controller = new AbortController();
                    const timeoutId = setTimeout(() => controller.abort(), 1000);
                    
                    const response = await fetch(`${ai}/${id}.txt`, { 
                        method: 'HEAD', // 只请求头部，不下载内容
                        signal: controller.signal 
                    });
                    
                    clearTimeout(timeoutId);
                    
                    if (response.ok) {
                        count = id; // 更新计数
                        consecutiveFailures = 0; // 重置连续失败计数
                    } else {
                        consecutiveFailures++;
                    }
                } catch (error) {
                    consecutiveFailures++;
                }
                
                // 如果连续5个ID都不存在，认为后面没有更多文章了
                if (consecutiveFailures >= 5) {
                    break;
                }
            }
            
            return count;
        }

        // 从指定分类加载文章
        async function loadArticlesFromai(ai, startId, endId) {
            if (endId < startId) return [];
            
            const articles = [];
            const cacheKeyPrefix = `${ai}_`;
            
            // 先检查缓存
            for (let id = startId; id <= endId; id++) {
                const cacheKey = cacheKeyPrefix + id;
                if (articleCache.has(cacheKey)) {
                    articles.push(articleCache.get(cacheKey));
                }
            }
            
            // 如果缓存中已有足够的文章，直接返回
            if (articles.length >= endId - startId + 1) {
                return articles;
            }
            
            // 逐个加载文章，避免请求过多
            for (let id = startId; id <= endId; id++) {
                const cacheKey = cacheKeyPrefix + id;
                if (articleCache.has(cacheKey)) {
                    continue; // 已缓存，跳过
                }
                
                try {
                    // 设置单篇文章加载超时
                    const controller = new AbortController();
                    const timeoutId = setTimeout(() => controller.abort(), 3000);
                    
                    const response = await fetch(`${ai}/${id}.txt`, { 
                        signal: controller.signal 
                    });
                    
                    clearTimeout(timeoutId);
                    
                    if (!response.ok) {
                        continue; // 文件不存在，继续下一个
                    }
                    
                    const arrayBuffer = await response.arrayBuffer();
                    const encoding = detectEncoding(arrayBuffer);
                    let text = '';
                    
                    if (encoding === 'utf-8') {
                        const decoder = new TextDecoder('utf-8');
                        text = decoder.decode(arrayBuffer);
                    } else {
                        const decoder = new TextDecoder('gbk');
                        text = decoder.decode(arrayBuffer);
                    }
                    
                    const lines = text.split('\n');
                    const title = lines[0] || `未命名文章 (${ai}-${id})`;
                    const content = lines.slice(1).join('\n');
                    
                    // 判断是否为Markdown文件
                    const isMarkdown = title.toLowerCase().endsWith('.md') || 
                                      (id.toString().includes('.md'));
                    
                    const article = {
                        id,
                        title: title.replace(/\.md$/i, ''),
                        content,
                        ai,
                        path: `${ai}/${id}.txt`,
                        isMarkdown
                    };
                    
                    articles.push(article);
                    articleCache.set(cacheKey, article); // 缓存文章
                } catch (error) {
                    if (error.name !== 'AbortError') {
                        console.error(`加载文章 ${ai}/${id}.txt 失败:`, error);
                    }
                    // 单篇文章加载失败不影响整体，继续加载下一篇
                    continue;
                }
            }
            
            return articles;
        }

        async function showArticleDetail(ai, id) {
            showLoading();

                const cacheKey = `${ai}_${id}`;
                let article;
                
                // 检查缓存
                if (articleCache.has(cacheKey)) {
                    article = articleCache.get(cacheKey);
                } else {

                    article = articles[0];
                    
                    if (!article) {
                        throw new Error('文章不存在或无法加载');

                }
                
                // 设置页面标题、描述和关键词
                setMetaTags(article);
                
                // 显示文章详情
                articleListContainer.classList.add('hidden');
                articleDetail.classList.remove('hidden');
                
                articleTitle.textContent = article.title;
                articleMeta.innerHTML = `
                    <span class="mr-4"><i class="fas fa-folder text-gray-400 mr-1"></i> ${aiNames[article.ai]}</span>

                `;

                setupImageErrorHandling();

            } catch (error) {
                console.error('显示文章详情失败:', error);
                showError('无法加载文章内容: ' + error.message);
            } finally {
                hideLoading();
            }
        }

        // 设置页面元标签
        function setMetaTags(article) {
            // 设置页面标题
            document.title = article.title || '鲁虺文本内容搜索展示系统';
            
            // 提取描述 - 从内容中随机取一段
            let description = '';
            const contentText = article.isMarkdown ? 
                article.content.replace(/#|_|\*|\[|\]|\(|\)/g, '') : 
                article.content;
                
            const sentences = contentText.split(/[。！？；,.!?;]/).filter(s => s.trim().length > 20);
            
            if (sentences.length > 0) {
                // 随机选择一段
                const randomIndex = Math.floor(Math.random() * sentences.length);
                description = sentences[randomIndex].trim().substring(0, 160); // 限制最大长度
            } else {
                // 如果没有合适的句子，使用标题
                description = article.title.substring(0, 160);
            }
            
            // 更新description元标签
            let metaDesc = document.querySelector('meta[name="description"]');
            if (metaDesc) {
                metaDesc.setAttribute('content', description);
            } else {
                metaDesc = document.createElement('meta');
                metaDesc.name = 'description';
                metaDesc.content = description;
                document.head.appendChild(metaDesc);
            }
            
            // 提取关键词
            let keywords = [];
            
            // 从标题提取关键词
            const titleWords = article.title.split(/\s+|，|,|；|;/).filter(w => w.length > 1);
            keywords.push(...titleWords.slice(0, 5)); // 最多取5个
            
            // 从内容提取关键词
            const contentWords = contentText.split(/\s+|，|,|；|;/).filter(w => w.length > 1);
            // 简单去重并随机选择一些词
            const uniqueWords = [...new Set(contentWords)];
            // 随机选择最多5个词
            for (let i = 0; i < Math.min(5, uniqueWords.length); i++) {
                const randomIndex = Math.floor(Math.random() * uniqueWords.length);
                keywords.push(uniqueWords[randomIndex]);
                uniqueWords.splice(randomIndex, 1); // 避免重复
            }
            
            // 去重并限制最多10个关键词
            keywords = [...new Set(keywords)].slice(0, 10);
            
            // 更新keywords元标签
            let metaKeywords = document.querySelector('meta[name="keywords"]');
            if (metaKeywords) {
                metaKeywords.setAttribute('content', keywords.join(','));
            } else {
                metaKeywords = document.createElement('meta');
                metaKeywords.name = 'keywords';
                metaKeywords.content = keywords.join(',');
                document.head.appendChild(metaKeywords);
            }
        }

        async function updateNavigationButtons(ai, currentId) {

            for (let id = currentId - 1; id >= 1; id--) {

                        prevId = id;
                        break;
                    }
                } catch (error) {
                    continue;
                }
            }
            
            // 查找下一篇文章
            let nextId = null;
            for (let id = currentId + 1; id <= currentId + 20; id++) { // 限制最多查找20个ID
                try {
                    const response = await fetch(`${ai}/${id}.txt`, { method: 'HEAD' });
                    if (response.ok) {
                        nextId = id;
                        break;
                    }
                } catch (error) {
                    continue;
                }
            }
            
            // 更新上一篇按钮
            if (prevId) {
                prevArticleButton.disabled = false;
                prevArticleButton.classList.remove('opacity-50', 'cursor-not-allowed');
                prevArticleButton.onclick = () => showArticleDetail(ai, prevId);
            } else {
                prevArticleButton.disabled = true;
                prevArticleButton.classList.add('opacity-50', 'cursor-not-allowed');
                prevArticleButton.onclick = null;
            }
            
            // 更新下一篇按钮
            if (nextId) {
                nextArticleButton.disabled = false;
                nextArticleButton.classList.remove('opacity-50', 'cursor-not-allowed');
                nextArticleButton.onclick = () => showArticleDetail(ai, nextId);
            } else {
                nextArticleButton.disabled = true;
