        function detectEncoding(buffer) {
            try {
                const detection = jschardet.detect(buffer);
                if (detection && detection.confidence > 0.5) {
                    state.detectedEncoding = detection.encoding.toLowerCase();
                    state.encodingConfidence = detection.confidence;
                    return state.detectedEncoding;
                }
            } catch (error) {
                console.error('编码检测失败:', error);
            }
            // 默认使用UTF-8
            state.detectedEncoding = 'utf8';
            state.encodingConfidence = 0;
            return 'utf8';
        }

        function convertToUtf8(content, encoding) {
            try {
                // 如果已经是UTF-8，直接返回
                if (encoding.toLowerCase() === 'utf8' || encoding.toLowerCase() === 'utf-8') {
                    return content;
                }
                
                // 尝试使用iconv-lite转换
                if (iconvLite.encodingExists(encoding)) {
                    return iconvLite.decode(Buffer.from(content, 'binary'), encoding);
                }
                
                // 特殊处理日语编码
                if (['shift-jis', 'euc-jp', 'iso-2022-jp'].includes(encoding.toLowerCase())) {
                    const encoded = new Uint8Array(content.length);
                    for (let i = 0; i < content.length; i++) {
                        encoded[i] = content.charCodeAt(i) & 0xFF;
                    }
                    return Encoding.convert(encoded, 'UTF8', encoding.toUpperCase());
                }
                
                // 尝试其他常见编码
                console.warn(`不支持的编码: ${encoding}，尝试自动转换`);
                return new TextDecoder(encoding, { fatal: false }).decode(Buffer.from(content, 'binary'));
            } catch (error) {
                console.error(`编码转换失败 (${encoding} to UTF-8):`, error);
                // 转换失败时返回原始内容的UTF-8表示
                return new TextDecoder('utf-8', { fatal: false }).decode(Buffer.from(content, 'binary'));
            }
        }

        async function loadArticleContent(articleId) {

                // 显示编码信息
