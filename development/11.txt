        function convertEncoding(content, inputEncoding = null) {
            try {
                // 如果未指定输入编码，则尝试自动检测
                if (!inputEncoding) {
                    // 对iso-8859-1有特殊检测逻辑
                    const isIso88591 = content.some(byte => byte > 127 && byte < 160);
                    if (isIso88591) {
                        inputEncoding = 'iso-8859-1';
                    } else {
                        // 尝试其他编码检测
                        for (const encoding of config.supportedEncodings) {
                            try {
                                if (encoding !== 'utf8' && encoding !== 'iso-8859-1') {
                                    const decoded = iconv.decode(Buffer.from(content), encoding);
                                    if (decoded.includes('�') === false) {
                                        inputEncoding = encoding;
                                        break;
                                    }
                                }
                            } catch (e) {
                                continue;
                            }
                        }
                        // 如果没有检测到合适的编码，默认使用utf8
                        inputEncoding = inputEncoding || 'utf8';
                    }
                }

                // 进行编码转换
                let decodedContent;
                if (inputEncoding === 'utf8') {
                    decodedContent = new TextDecoder().decode(Uint8Array.from(content));
                } else {
                    // 使用iconv-lite进行转换
                    decodedContent = iconv.decode(Buffer.from(content), inputEncoding);
                }

                // 处理可能的特殊字符
                return decodedContent
                    .replace(/\r\n/g, '\n')
                    .replace(/\r/g, '\n')
                    .replace(/�/g, '�'); // 替换无法识别的字符为通用替换符号
            } catch (error) {
                console.error('编码转换失败:', error);
                // 转换失败时，尝试使用文本解码器的替换模式
                return new TextDecoder('utf-8', { fatal: false, ignoreBOM: true }).decode(Uint8Array.from(content));
            }
        }

        async function loadTxtFile(url) {
            try {
                const response = await fetch(url);
                if (!response.ok) {
                    throw new Error(`HTTP错误: ${response.status}`);
                }
                
                // 获取原始二进制数据
                const arrayBuffer = await response.arrayBuffer();
                const bytes = new Uint8Array(arrayBuffer);
                
                // 尝试多种编码进行解码
                let content = null;
                let lastError = null;
                
                // 优先尝试utf8
                try {
                    content = new TextDecoder('utf-8', { fatal: true }).decode(bytes);
                } catch (e) {
                    lastError = e;
                }
                
                // 如果utf8解码失败，尝试其他编码
                if (!content) {
                    for (const encoding of config.supportedEncodings) {
                        if (encoding === 'utf8') continue;
                        
                        try {
                            content = convertEncoding(bytes, encoding);
                            // 简单验证解码质量

                            }

                            lastError = e;

                        }
                    }

                // 如果所有编码都失败，使用最后一次尝试的结果或原始数据

1. **多编码支持**：
   - 支持自动检测文件编码
   - 手动选择包括iso-8859-1在内的多种编码格式（UTF-8、GBK、GB2312等）
   - 确保不同编码的TXT文件在UTF-8网页上正确显示，避免乱码

2. **完整的数据库功能**：
   - 加载TXT文件作为数据库（支持CSV格式的TXT文件）
   - 查看、搜索、筛选数据记录
   - 添加、编辑、删除记录
   - 保存修改到原始编码文件
   - 导出为UTF-8编码文件

3. **用户体验优化**：
   - 右侧固定的快速返回顶部按钮
   - 右侧固定的快速前往底部按钮
   - 按钮根据滚动位置智能显示/隐藏
   - 响应式设计，适配不同屏幕尺寸
   - 操作反馈和状态提示

4. **数据安全**：
   - 操作前确认
   - 详细的错误提示
   - 成功操作反馈

使用方法很简单：选择一个TXT文件（CSV格式），选择适当的编码（或使用自动检测），设置字段分隔符，然后加载文件即可开始管理数据。系统会自动处理编码转换，确保在网页上正确显示各种编码的内容。
